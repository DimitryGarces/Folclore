CREATE TABLE ROLE (Id_Rol Int(5) PRIMARY KEY AUTO_INCREMENT, Name Varchar(30) NOT NULL);

CREATE TABLE USER (Id_User Int(5) PRIMARY KEY AUTO_INCREMENT, 
Id_Rol INT(5), FOREIGN KEY (Id_Rol) REFERENCES ROLE(Id_Rol), Name Varchar(30) NOT NULL, 
LastName Varchar(30), UserName Varchar(30) NOT NULL,
Password Varchar(30) NOT NULL, Age Int(3) NOT NULL);


CREATE TABLE STATE (Id_State INT(5) PRIMARY KEY AUTO_INCREMENT,
Name VARCHAR(30) NOT NULL);

CREATE TABLE COSTUME(Id_Costume INT(5) PRIMARY KEY AUTO_INCREMENT, 
Name VARCHAR(30) NOT NULL, Path VARCHAR(60) NOT NULL);

CREATE TABLE VIDEO (Id_Video INT(5) PRIMARY KEY AUTO_INCREMENT, Id_State INT(5),
FOREIGN KEY (Id_State) REFERENCES STATE(Id_State),
Title VARCHAR(30) NOT NULL,
Author VARCHAR(20) NOT NULL,
Path VARCHAR (60) NOT NULL, 
DatePublic DATE,
Description VARCHAR(60));

CREATE TABLE COMMENT (Id_Comment Int (5) PRIMARY KEY AUTO_INCREMENT,
Id_User INT(5), FOREIGN KEY (Id_User) REFERENCES USER(Id_User),
Id_Video INT(5), FOREIGN KEY (Id_Video) REFERENCES VIDEO(Id_Video),
DatePublic DATE,
Content VARCHAR(100));

CREATE TABLE VIDEOCOSTUME(Id_Video INT(5), Id_Costume INT(5), Description VARCHAR(60),
FOREIGN KEY (Id_Video) REFERENCES VIDEO(Id_Video),
FOREIGN KEY (Id_Costume) REFERENCES COSTUME(Id_Costume));
